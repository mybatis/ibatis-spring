25  /**
26   * Additional tests for {@link AbstractLobTypeHandler} covering error paths not exercised by {@link LobTypeHandlerTests}
27   * (which has an {@code @AfterEach} that verifies LobCreator.close()).
28   */
29  @SuppressWarnings("deprecation")
30  class AbstractLobTypeHandlerAdditionalTests {
31  
32    // ---- Constructor guards ----
33  
34    @Test
35    void testConstructorWithNullLobHandlerThrowsIllegalStateException() {
36      assertThrows(IllegalStateException.class, () -> new ClobStringTypeHandler(null));
37    }
38  
39    @Test
40    void testDefaultConstructorWithNoConfigTimeLobHandlerThrows() {
41      // getConfigTimeLobHandler() returns null outside of factory-bean initialisation, so the
42      // default no-arg constructor must throw IllegalStateException.
43      assertThrows(IllegalStateException.class, ClobStringTypeHandler::new);
44    }
45  
46    // ---- setParameter without active transaction synchronization ----
47  
48    @Test
49    void testSetParameterWithoutSynchronizationThrowsIllegalStateException() {
50      LobHandler lobHandler = mock(LobHandler.class);
51      ClobStringTypeHandler type = new ClobStringTypeHandler(lobHandler);
52      PreparedStatement ps = mock(PreparedStatement.class);
53  
54      assertFalse(TransactionSynchronizationManager.isSynchronizationActive(),
55          "Synchronization must not be active for this test");
56  
57      assertThrows(IllegalStateException.class, () -> type.setParameter(ps, 1, "content", null));
58    }
59  
60    // ---- getResult(CallableStatement, int) ----
61  
62    @Test
63    void testGetResultFromCallableStatementThrowsSqlException() {
64      LobHandler lobHandler = mock(LobHandler.class);
65      ClobStringTypeHandler type = new ClobStringTypeHandler(lobHandler);
66      CallableStatement cs = mock(CallableStatement.class);
67  
68      assertThrows(SQLException.class, () -> type.getResult(cs, 1));
69    }
70  
71    // ---- resolveJdbcType utility method ----
72  
73    @Test
74    void testResolveJdbcTypeWithNullReturnsEmptyString() {
75      LobHandler lobHandler = mock(LobHandler.class);
76      ClobStringTypeHandler type = new ClobStringTypeHandler(lobHandler);
77      assertEquals("", type.resolveJdbcType(null));
78    }
79  
80    @Test
81    void testResolveJdbcTypeWithValueReturnsValue() {
82      LobHandler lobHandler = mock(LobHandler.class);
83      ClobStringTypeHandler type = new ClobStringTypeHandler(lobHandler);
84      assertEquals("CLOB", type.resolveJdbcType("CLOB"));
85    }
86  
87    // ---- IOException in setParameter translated to SQLException ----
88  
89    @Test
90    void testSetParameterIOExceptionWrappedAsSqlException() throws Exception {
91      LobHandler lobHandler = mock(LobHandler.class);
92      LobCreator lobCreator = mock(LobCreator.class);
93      org.mockito.BDDMockito.given(lobHandler.getLobCreator()).willReturn(lobCreator);
94  
95      PreparedStatement ps = mock(PreparedStatement.class);
96      // Object is not Serializable, so serialization throws NotSerializableException (IOException subtype)
97      Object nonSerializable = new Object();
98  
99      BlobSerializableTypeHandler type = new BlobSerializableTypeHandler(lobHandler);
100     TransactionSynchronizationManager.initSynchronization();
101     try {
102       assertThrows(SQLException.class, () -> type.setParameter(ps, 1, nonSerializable, null));
103     } finally {
104       TransactionSynchronizationManager.clearSynchronization();
105     }
106   }
107 
108   // ---- IOException in getResult translated to SQLException ----
109 
110   @Test
111   void testGetResultIOExceptionWrappedAsSqlException() throws Exception {
112     LobHandler lobHandler = mock(LobHandler.class);
113     ResultSet rs = mock(ResultSet.class);
114     // Corrupt bytes – not a valid Java serialization stream
115     byte[] invalidData = { 1, 2, 3 };
116     org.mockito.BDDMockito.given(lobHandler.getBlobAsBinaryStream(rs, 1))
117         .willReturn(new ByteArrayInputStream(invalidData));
118 
119     BlobSerializableTypeHandler type = new BlobSerializableTypeHandler(lobHandler);
120     assertThrows(SQLException.class, () -> type.getResult(rs, 1));
121   }

123 }
