7   package org.springframework.orm.ibatis.support;
8   
9   import java.sql.PreparedStatement;
10  import java.sql.ResultSet;
11  import java.sql.SQLException;
12  
13  import org.springframework.jdbc.support.lob.LobCreator;
14  import org.springframework.jdbc.support.lob.LobHandler;
15  
16  /**
17   * iBATIS TypeHandler implementation for byte arrays that get mapped to BLOBs. Retrieves the LobHandler to use from
18   * SqlMapClientFactoryBean at config time.
19   * <p>
20   * Can also be defined in generic iBATIS mappings, as DefaultLobCreator will work with most JDBC-compliant database
21   * drivers. In this case, the field type does not have to be BLOB: For databases like MySQL and MS SQL Server, any large
22   * enough binary type will work.
23   *
24   * @author Juergen Hoeller
25   *
26   * @since 1.1.5
27   *
28   * @see org.springframework.orm.ibatis.SqlMapClientFactoryBean#setLobHandler
29   *
30   * @deprecated as of Spring 3.2, in favor of the native Spring support in the Mybatis follow-up project
31   *             (https://mybatis.org/)
32   */
33  @Deprecated
34  public class BlobByteArrayTypeHandler extends AbstractLobTypeHandler {
35  
36    /**
37     * Constructor used by iBATIS: fetches config-time LobHandler from SqlMapClientFactoryBean.
38     *
39     * @see org.springframework.orm.ibatis.SqlMapClientFactoryBean#getConfigTimeLobHandler
40     */
41    public BlobByteArrayTypeHandler() {
42      super();
43    }
44  
45    /**
46     * Constructor used for testing: takes an explicit LobHandler.
47     *
48     * @param lobHandler
49     *          the LobHandler to use
50     */
51    protected BlobByteArrayTypeHandler(LobHandler lobHandler) {
52      super(lobHandler);
53    }
54  
55    @Override
56    protected void setParameterInternal(PreparedStatement ps, int index, Object value, String jdbcType,
57        LobCreator lobCreator) throws SQLException {
58      lobCreator.setBlobAsBytes(ps, index, (byte[]) value);
59    }
60  
61    @Override
62    protected Object getResultInternal(ResultSet rs, int index, LobHandler lobHandler) throws SQLException {
63      return lobHandler.getBlobAsBytes(rs, index);
64    }
65  
66    @Override
67    public Object valueOf(String s) {
68      return s.getBytes();
69    }
70  
71  }
